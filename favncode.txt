    <script>
        const spotlightContainer = document.getElementById('spotlight-container');
        const gridContainer = document.getElementById('matches-grid');
        const DATA_URL = 'https://raw.githubusercontent.com/Jitendra-unatti/fancode/refs/heads/main/data/fancode.json';
        const PLAY_BASE = 'fc/player.html';

        async function fetchAndRender() {
            try {
                const res = await fetch(DATA_URL);
                const data = await res.json();
                let matches = Array.isArray(data.matches) ? data.matches : [];

                matches.sort((a, b) => {
                    const aIsLive = a.status?.toUpperCase() === 'LIVE';
                    const bIsLive = b.status?.toUpperCase() === 'LIVE';
                    return bIsLive - aIsLive;
                });

                gridContainer.innerHTML = '';

                const firstLiveMatch = matches.find(m => m.status?.toUpperCase() === 'LIVE');
                if (firstLiveMatch) {
                    renderSpotlight(firstLiveMatch);
                    matches = matches.filter(m => m.match_id !== firstLiveMatch.match_id);
                }

                if (matches.length > 0) {
                    matches.forEach((match, index) => {
                        const card = createMatchCard(match);
                        card.style.animationDelay = `${index * 100}ms`;
                        gridContainer.appendChild(card);
                    });
                } else if (!firstLiveMatch) {
                    gridContainer.innerHTML = `<p class="col-span-full text-center text-slate-400">No matches available right now.</p>`;
                }
                
                initInteractiveEffects();

            } catch (error) {
                console.error("Failed to load match data:", error);
                gridContainer.innerHTML = `<p class="col-span-full text-center text-red-400">Could not load matches. Please check your connection and try again.</p>`;
            }
        }

        function renderSpotlight(match) {
            const imgSrc = match.image || 'https://fancode.com/skillup-uploads/cms-media/Cricket_Fallback_Old_match-card.jpg';
            const buttonsHTML = generateButtons(match, 'text-sm px-5 py-2.5 rounded-lg');

            spotlightContainer.innerHTML = `
                <div class="spotlight-card rounded-2xl overflow-hidden grid grid-cols-1 lg:grid-cols-2 items-center gap-8 p-6 md:p-8">
                    <div class="relative rounded-lg overflow-hidden aspect-video">
                        <img src="${imgSrc}" alt="${match.title}" class="w-full h-full object-cover">
                        <div class="absolute inset-0 bg-gradient-to-t from-black/50 to-transparent"></div>
                        <div class="spotlight-live-badge absolute top-4 left-4 text-xs font-bold px-3 py-1.5 rounded-full text-white shadow-lg">
                            <i class="fas fa-circle text-xs mr-1.5 animate-ping absolute opacity-75"></i>
                            <i class="fas fa-circle text-xs mr-1.5"></i>
                            LIVE NOW
                        </div>
                    </div>
                    <div class="flex flex-col justify-center">
                        <p class="text-amber-400 font-semibold text-sm mb-1">${match.tournament || ''}</p>
                        <h2 class="text-2xl md:text-3xl font-bold text-white mb-3">${match.title || 'Live Match'}</h2>
                        <p class="text-slate-400 text-sm mb-6">${match.language || ''}</p>
                        <div class="flex flex-wrap gap-3">${buttonsHTML}</div>
                    </div>
                </div>
            `;
        }

        function createMatchCard(match) {
            const card = document.createElement('div');
            card.className = 'match-card rounded-xl overflow-hidden flex flex-col';
            
            const isLive = match.status?.toUpperCase() === 'LIVE';
            const imgSrc = match.image || 'https://fancode.com/skillup-uploads/cms-media/Cricket_Fallback_Old_match-card.jpg';
            const buttonsHTML = generateButtons(match, 'text-xs px-3 py-2 rounded-md');

            card.innerHTML = `
                <div class="relative">
                    <div class="card-glow"></div>
                    <img src="${imgSrc}" alt="${match.title}" class="w-full h-48 object-cover">
                    ${isLive ? `<span class="absolute top-3 left-3 bg-red-600 text-white text-xs font-semibold px-3 py-1 rounded-full shadow-md">LIVE</span>` : ''}
                </div>
                <div class="p-5 flex flex-col flex-grow">
                    <p class="text-amber-500 text-xs font-semibold mb-1">${match.tournament || ''}</p>
                    <h3 class="text-lg font-bold text-slate-100 flex-grow mb-4">${match.title || 'Match Title'}</h3>
                    <div class="flex flex-wrap gap-2">${buttonsHTML || '<p class="text-xs text-slate-500">Links coming soon...</p>'}</div>
                </div>
            `;
            return card;
        }

        function generateButtons(match, classes) {
            const streams = {
                ...match.STREAMING_CDN,
                adfree_stream: match.adfree_stream,
                dai_stream: match.dai_stream
            };
            
            let buttons = '';
            const id = encodeURIComponent(String(match.match_id ?? ''));

            for (const [key, value] of Object.entries(streams)) {
                // MODIFICATION: Added a check to skip 'Primary_Playback_URL'
                if (value && value !== "Unavailable" && key.toLowerCase() !== 'language' && key !== 'Primary_Playback_URL') {
                    const label = key.replace(/_/g, ' ').replace(/(cdn|stream)/i, '').trim().toUpperCase() || 'STREAM';
                    const href = `${PLAY_BASE}?id=${id}&cdn=${encodeURIComponent(key)}`;
                    buttons += `<a href="${href}" class="watch-btn text-center text-black font-semibold ${classes}">${label}</a>`;
                }
            }
            return buttons;
        }

        function initInteractiveEffects() {
            const cards = document.querySelectorAll('.match-card');
            cards.forEach(card => {
                const glow = card.querySelector('.card-glow');
                card.addEventListener('mousemove', (e) => {
                    const rect = card.getBoundingClientRect();
                    const x = e.clientX - rect.left;
                    const y = e.clientY - rect.top;
                    
                    const rotateX = (y - rect.height / 2) / -15;
                    const rotateY = (x - rect.width / 2) / 15;
                    card.style.transform = `perspective(1000px) rotateX(${rotateX}deg) rotateY(${rotateY}deg) scale3d(1.02, 1.02, 1.02)`;
                    
                    glow.style.setProperty('--x', `${x}px`);
                    glow.style.setProperty('--y', `${y}px`);
                });

                card.addEventListener('mouseleave', () => {
                    card.style.transform = 'perspective(1000px) rotateX(0) rotateY(0) scale3d(1, 1, 1)';
                });
            });
        }
        
        function handlePopup() {
            const popup = document.getElementById('telegram-popup');
            const closeBtn = document.getElementById('close-popup-btn');
            const joinBtn = document.getElementById('join-btn');
            const POPUP_KEY = 'telegramPopupDismissed';

            if (localStorage.getItem(POPUP_KEY)) {
                return;
            }

            setTimeout(() => {
                popup.classList.add('show');
            }, 3000);

            function dismissPopup() {
                popup.classList.remove('show');
                localStorage.setItem(POPUP_KEY, 'true');
            }

            closeBtn.addEventListener('click', dismissPopup);
            joinBtn.addEventListener('click', () => {
                setTimeout(dismissPopup, 500); 
            });
        }

        document.addEventListener('DOMContentLoaded', () => {
            fetchAndRender();
            handlePopup();
        });
  
    </script>
